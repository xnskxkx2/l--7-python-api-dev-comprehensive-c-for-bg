def find_post(post_id: int, posts: List[Dict[str, Any]]) -> Optional[Dict[str, Any]]:
    return next((p for p in posts if p.get("id") == post_id), None)


def find_index_post(post_id: int, posts: List[Dict[str, Any]]) -> Optional[int]:
    return next((i for i, post in enumerate(posts) if post.get("id") == post_id), None)
